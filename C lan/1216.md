#include <stdio.h>
#include <string.h>

int main(){
	char text[80];
	printf("輸入字串: ");
	scanf("%s",text);
	int i=0,a=0,p=0;
	while(text[i]!='\0'){
		if(text[i]=='a')
			a++;
		else if(text[i]=='p')
			p++;
		
		i++;
	}
	printf("輸出結果 %s\n",text);
	printf("a 字元共有 %d\n",a);
	printf("p 字元共有 %d\n",p);
	


}
