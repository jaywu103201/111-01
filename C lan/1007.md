變數
===
```
int main (){
	int age;
	double score;
	char level='?';
	printf("\n年級\t分數\t級別\n");
	printf("%d\t%.2f\t%c\n",age,score,level);
	
	age=5;
	score=80.0;
	level='B';
	printf("\n年級\t分數\t級別\n");
	printf("%d\t%.2f\t%c\n",age,score,level);	
}
```
